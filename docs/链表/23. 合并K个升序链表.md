# [23] 合并K个升序链表



```java
    public ListNode mergeKLists(ListNode[] lists) {
        if(lists == null || lists.length == 0){
            return null;
        }

        int k = lists.length;

        ListNode head = new ListNode(-1);
        ListNode p = head;

        // 一个数组记录每个子序列的游标
        ListNode[] cur = new ListNode[k];

        // 难点在于每次找到k个元素中的最小值，这里用最小堆实现
        // 第一个参数是初始容量，第二个参数是比较器，升序就是构造最小堆
        PriorityQueue<ListNode> pq = new PriorityQueue<ListNode>(k, (a, b) -> a.val - b.val);

        // 初始化：每个子序列都放当前第一个元素进去
        for(int i = 0; i < k; i++){
            ListNode n = lists[i];
            if(n == null){
                continue;
            }

            pq.offer(n);
            cur[i] = n;
        }

        for(int j = 0; j < 10000 && pq.size() > 0; j++){
            // 找到k个元素中最小的
            ListNode min = pq.poll();
            p.next = min;
            p = min;

            // 找出这个最小元素是哪个子序列的，然后子序列向前推进
            for(int i = 0; i < k; i++){
                ListNode n = cur[i];
                if(n == null){
                    continue;
                }

                if(min == n){
                    n = n.next;
                    // 如果某个子序列没有数据了，减少参与合并的数量
                    if(n != null){
                        pq.offer(n);
                    }

                    cur[i] = n;
                    break;
                }
            }
        }

        return head.next;
    }
```

```java
    public ListNode mergeKLists(ListNode[] lists) {
        if(lists == null || lists.length == 0){
            return null;
        }

        int k = lists.length;

        ListNode head = new ListNode(-1);
        ListNode p = head;

        // 难点在于每次找到k个元素中的最小值，这里用最小堆实现
        // 第一个参数是初始容量，第二个参数是比较器，升序就是构造最小堆
        PriorityQueue<ListNode> pq = new PriorityQueue<ListNode>(k, (a, b) -> a.val - b.val);

        // 初始化：每个子序列都放当前第一个元素进去
        for(ListNode n : lists){
            if(n != null){
                pq.offer(n);
            }
        }

        // 只要堆里还有数据说明还没取完
        while(!pq.isEmpty()){
            // 找到k个元素中最小的
            ListNode min = pq.poll();
            p.next = min;
            p = min;

            // 如果当前序列还有下一个则放入下一个
            if(min.next != null){
                pq.offer(min.next);
            }
        }

        return head.next;
    }
```
