# 645.错误的集合

# 排序

    public int[] findErrorNums(int[] nums) {
        Arrays.sort(nums);

        int dup = 0, lost = 1;
        for(int i = 0; i < nums.length; i++){
            if(i > 0 && nums[i-1] == nums[i]){
                dup = nums[i];
            }

            // 只要lost与当前num相等，lost就会自增
            // 如果下一个num不出现lost就会停留在确实的数字的位置
            if(nums[i] == lost){
                lost++;
            }
        }

        return new int[]{dup, lost};
    }

# 求余运算

    public int[] findErrorNums(int[] nums) {
        int n = nums.length;
        boolean[] flag = new boolean[n];

        int dup = 0, lost = 0;
        for(int i = 0; i < n; i++){
            // 求余运算后余数作为index就是有序的了
            int idx = nums[i] % n;
            if(flag[idx]){
                dup = nums[i];
            }
            
            flag[idx] = true;
        }

        for(int i = 0; i < n; i++){
            if(!flag[i]){
                // 对0位置单独判断是因为0代表整除了
                lost = i == 0 ? n : i;
                break;
            }
        }

        return new int[]{dup, lost};
    }

# 原地相反数O(1)空间

思路和求余相似，在nums数组基础上使用相反数来实现，不需要额外的空间

    public int[] findErrorNums(int[] nums) {
        int n = nums.length;
        int dup = 0, lost = 0;
        for(int i = 0; i < n; i++){
            // 将i所在位置的值直接作为index
            int idx = Math.abs(nums[i])-1;

            // 如果出现过两次的值，作为索引必然取到负值
            if(nums[idx] < 0){
                dup = Math.abs(nums[i]);
            }else{
                // 出现过得索引，改为相反数
                nums[idx] = -nums[idx];
            }
        }

        for(int i = 0; i < n; i++){
            // 大于0说明没有出现过该值，就是丢失的数
            if(nums[i] > 0){
                lost = i + 1;
            }
        }

        return new int[]{dup, lost};
    }
