# 419.甲板上的战舰

# 第一次实现

    public int countBattleships(char[][] board) {
        int m = board.length, n = board[0].length;
        int count = 0;
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                // 上方有单元格
                if(i-1 >= 0){
                    // 当前单元格有战舰且上方必须是空位才能作为计数前提
                    if(board[i-1][j] == '.' && board[i][j] == 'X'){
                        // 判断上方后还要判断左方必须是空位才行
                        if(j-1 < 0 || board[i][j-1] == '.'){
                            count++;   
                        }
                    }
                }
                // 上方无单元格(第一行)
                else{
                    // 当前单元格有战舰
                    if(board[i][j] == 'X'){
                        // 因为是第一行所以只需要判断左方必须是空位
                        if(j-1 < 0 || board[i][j-1] == '.'){
                            count++;
                        }
                    }
                }
            }
        }

        return count;
    }

# 优化后(枚举起点)

根据第一次实现的思路，其实当单元格是X时，只需要判断左方和上方两个单元格是否是空位，只有当[i,j]的左侧和上方都是空位才能记为新战舰

    public int countBattleships(char[][] board) {
        int m = board.length, n = board[0].length;
        int count = 0;
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                // 只有当前单元格是X才需要判断是否进行计数
                if(board[i][j] == 'X'){
                    // 当[i,j]的左侧和上方都是空位，就可以记为新战舰了
                    if((i-1 < 0 || board[i-1][j] == '.') && (j-1 < 0 || board[i][j-1] == '.')){
                         count++;   
                    }
                }
                
            }
        }

        return count;
    }
