# 43.字符串相乘

# 模拟(一)

1、先求num2的每一位数与num1的乘积 2、将乘积全部相加，注意补零

    //1、先求num2的每一位数与num1的乘积
    //2、将乘积全部相加，注意补零
    public String multiply(String num1, String num2) {
        // 0乘以任何数都是0, 直接返回
        if(num1.charAt(0) == '0' || num2.charAt(0) == '0'){
            return "0";
        }
        
        // 记录num2的每一位数与num1的乘积
        List<String> mul_list = new LinkedList<>();

        // 记录最长乘积的长度，用于下文加法
        int n = 0;

        // 计算出num2的每一位数与num1的乘积     
        int n2 = num2.length();
        for(int i = 0; i < n2; i++){
            String mul_str = multiply(num1, num2.charAt(n2-1-i), i);

            n = Math.max(n, mul_str.length());

            mul_list.add(mul_str);
        }

        //System.out.println("n:"+n+"mul_list:"+mul_list);

        // 用一个变量记录进位
        int carry = 0;
        // 记录加法求和的结果
        StringBuilder sb = new StringBuilder();
        // 遍历所有乘积开始相加
        for(int i = 0; i < n; i++){
            for(String str : mul_list){
                // 对每一个参与求和的字符串如果找不到字符就补0
                carry += str.length() > i ? str.charAt(str.length()-1-i) - '0' : 0;
            }

            // 计算当前位值
            sb.append(carry % 10);

            // 计算进位值
            carry /= 10;
        }

        // 如果最后还有进位说明溢出总长度
        if(carry > 0){
            sb.append(carry);
        }

        sb.reverse();

        return sb.toString();
    }

    // 求num2的每一位数与num1的乘积
    private String multiply(String num1, char ch_of_num2, int idx) {
        // 0乘以任何数都是0, 直接返回
        if(ch_of_num2 == '0'){
            return "0";
        }

        // 用一个变量记录进位
        int carry = 0;

        // 记录加法求和的结果
        StringBuilder sb = new StringBuilder();

        // 求乘积
        int n = num1.length();
        for(int i = 0; i < n; i++){
            carry += (num1.charAt(n-1-i) - '0') * (ch_of_num2 - '0');

            // 计算当前位值
            sb.append(carry % 10);

            // 计算进位值
            carry /= 10;
        }

        if(carry > 0){
            sb.append(carry);
        }

        sb.reverse();

        // 这里注意，需要对结果补零，后续的加法才能对齐
        for(int i = 0; i < idx; i++){
            sb.append('0');
        }

        return sb.toString();
    }

# 模拟(二)

有两个指针 i，j 在 num1 和 num2 上游走，计算乘积，同时将乘积叠加到 res 数组的正确位置：

细心观察之后就发现，num1[i] 和 num2[j] 的乘积对应的就是 res[i+j] 和 res[i+j+1] 这两个位置。

    public String multiply(String num1, String num2) {
        if(num1.charAt(0) == '0' || num2.charAt(0) == '0'){
            return "0";
        }

        int n1 = num1.length(), n2 = num2.length();
        int[] res = new int[n1+n2];

        // 模拟乘法，从最右边开始
        for(int i = n1-1; i >= 0 ; i--){
            for(int j = n2-1; j >= 0; j--){
                int mul = (num1.charAt(i) - '0') * (num2.charAt(j) - '0');

                // 求和
                mul += res[i+j+1];

                // 求值和进位
                res[i+j+1] = mul % 10;
                res[i+j] += mul / 10;
            }
        }

        //System.out.println("res:" + Arrays.toString(res));

        // 找到第一个不为零的数字就是开始
        int start = 0;
        while(res[start] <= 0){
            start++;
        }

        //System.out.println("start:" + start);

        StringBuilder sb = new StringBuilder();
        for(int i = start; i < res.length; i++){
            sb.append((char)(res[i] + '0'));
        }

        return sb.toString();
    }
